package main 
 
import "fmt" 
 
type student struct { 
 Name  string 
 Age   int
 Grade string 
} 

func (s *student) show() { 
fmt.Printf("Name: %s\n", s.Name) 
fmt.Printf("Age: %d\n", s.Age) 
fmt.Printf("Grade: %s\n", s.Grade) 
} 
func main() { 
s := &student{ 
Name:  "John Doe", 
Age:   18, 
Grade: "A", 
} 
s.show() 
}
