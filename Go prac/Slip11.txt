package main
import (
    "fmt"
)

func main() {
    var num int

    fmt.Println("Enter a number:")
    fmt.Scanln(&num)

    if num >= 10 && num <= 99 {
        fmt.Println("The entered number is a two-digit number.")
    } else {
        fmt.Println("The entered number is not a two-digit number.")
    }
}
